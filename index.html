# SISTEMA CADASTRO

usuarios = {}

print("=== CADASTRAR-SE ===")

while True:
    nome = input("Nome: ")
    if nome.strip() == "":
        print("Nome inválido")
        continue
    break

while True:
    senha = input("Senha: ")
    if senha.strip() == "" or len(senha) < 4:
        print("Senha inválida")
        continue
    break

while True:
    confirmar = input("Confirme: ")
    if senha != confirmar:
        print("Não coincide")
        continue
    break

usuarios[nome] = senha
print("Cadastro finalizado")

while True:
    nome_login = input("Nome: ")
    if nome_login in usuarios:
        break

while True:
    senha_login = input("Senha: ")
    if usuarios[nome_login] == senha_login:
        print("Login sucesso")
        break
